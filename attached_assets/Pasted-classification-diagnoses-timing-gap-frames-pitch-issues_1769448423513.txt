A) Contact Efficiency Scoring Function (Python)
def calculate_contact_efficiency(
    game_readiness_score,
    classification,
    diagnoses,
    timing_gap_frames,
    pitch_issues
):
    # Internal only score (never shown directly)
    ces = 100

    # Core caps
    if classification == "early_commit":
        ces -= 30

    if classification == "arm_dominant_swing":
        ces -= 35

    if "head_drift_balance_loss" in diagnoses:
        ces -= 20

    if timing_gap_frames and timing_gap_frames > 10:
        ces -= 15

    if classification == "simultaneous_start":
        ces -= 10

    # Pitch-context caps
    for issue in pitch_issues:
        if issue in ["fastball", "breaking"]:
            ces -= 10

    # Readiness coupling (hard caps)
    if game_readiness_score < 40:
        ces = min(ces, 50)
    elif game_readiness_score < 60:
        ces = min(ces, 70)

    # Clamp
    ces = max(0, min(100, ces))

    return ces

B) Map CES → User-Visible Range (Apple-Safe)
def contact_efficiency_label(ces):
    if ces <= 30:
        return "Low"
    elif ces <= 55:
        return "Low–Moderate"
    elif ces <= 75:
        return "Moderate"
    elif ces <= 90:
        return "Moderate–High"
    else:
        return "High"
